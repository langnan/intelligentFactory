<template>
  <div class="app-container">
    <div class="header"><b>《大风车》客户端</b> 版本列表</div>
    <Table
      :pagination="false"
      :order-number="false"
      :header-cell-style="headerCellStyle"
      :table-json="tableJson"
      :table-data="tableData"
    />
  </div>
</template>

<script>
import Table from '@/components/CommonTable/CommonTable.vue'

  name: 'VersionManagement',

    Table
